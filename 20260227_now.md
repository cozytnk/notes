
```js
const now = new Date().toLocaleString().replace(/\d+/g, m => m.padStart(2, '0')).replace(/[/:]/g, '').replace(/ /g, 'T')
// '2025/8/14 9:28:20' -> '20250814T092820'

const now = new Date().toLocaleString('ja', { dateStyle: 'medium', timeStyle: 'medium' }).replace(/[/:]/g, '').replace(/ /g, 'T')
// '2025/08/14 9:22:16' -> '20250814T92216'

const now = new Date().toLocaleString('ja', { dateStyle: 'medium' }).replace(/\//g, '')
// '2025/08/13' -> '20250813'
```
